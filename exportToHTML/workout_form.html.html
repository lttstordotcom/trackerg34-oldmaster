<script th:inline="javascript">
    // If we're editing, we may receive intervalData from controller
    const intervalData = /*[[${intervalData}]]*/ null;

    function showIntervals(show) {
        const box = document.getElementById('intervalsBox');
        box.style.display = show ? 'block' : 'none';
    }

    function addIntervalRow(prefill) {
        const rows = document.getElementById('intervalRows');
        const countEl = document.getElementById('intervalCount');
        let idx = parseInt(countEl.value, 10);

        const row = document.createElement('div');
        row.className = 'intervalRow';
        row.dataset.idx = idx;

        row.innerHTML = `
      <div> 
        <label>Work distance (m)</label> 
        <input type="number" min="0" name="intervals[${idx}].workDistanceMeters" placeholder="e.g., 500" 
               value="${prefill?.workDistanceMeters ?? ''}"> 
      </div> 
 
      <div> 
        <label>Work time (min:sec)</label> 
        <div class="inline"> 
          <input class="small" type="number" min="0" name="intervals[${idx}].workMinutes" placeholder="min" 
                 value="${prefill ? Math.floor((prefill.workTimeSeconds||0)/60) : ''}"> 
          <input class="small" type="number" min="0" name="intervals[${idx}].workSeconds" placeholder="sec" 
                 value="${prefill ? ((prefill.workTimeSeconds||0)%60) : ''}"> 
        </div> 
      </div> 
 
      <div> 
        <label>Rest (min:sec)</label> 
        <div class="inline"> 
          <input class="small" type="number" min="0" name="intervals[${idx}].restMinutes" placeholder="min" 
                 value="${prefill ? Math.floor((prefill.restTimeSeconds||0)/60) : ''}"> 
          <input class="small" type="number" min="0" name="intervals[${idx}].restSeconds" placeholder="sec" 
                 value="${prefill ? ((prefill.restTimeSeconds||0)%60) : ''}"> 
        </div> 
      </div> 
 
      <div> 
        <button class="smallBtn" type="button" onclick="removeRow(${idx})">X</button> 
      </div> 
    `;

        rows.appendChild(row);
        idx++;
        countEl.value = idx;
    }

    function removeRow(idx) {
        const rows = document.getElementById('intervalRows');
        const node = rows.querySelector(`[data-idx="${idx}"]`);
        if (node) node.remove();
    }

    // Toggle logic
    const toggle = document.getElementById('intervalToggle');
    toggle.addEventListener('change', () => {
        showIntervals(toggle.checked);
        if (toggle.checked && document.getElementById('intervalCount').value === "0") {
            addIntervalRow();
        }
    });

    // On page load: set interval box if checked OR editing with interval data
    window.addEventListener('load', () => {
        showIntervals(toggle.checked);

        // If editing and we have interval data, render rows
        if (intervalData && Array.isArray(intervalData) && intervalData.length > 0) {
            toggle.checked = true;
            showIntervals(true);
            intervalData.forEach(it => addIntervalRow(it));
        } else if (toggle.checked) {
            addIntervalRow();
        }
    });
</script>
